-- Realizar una consulta que permita conocer en que zapatos fue usado determinado molde.
SELECT 
    z.id_zapato, 
    d.nombre_diseno,
    z.id_molde
FROM zapato z
JOIN lote_produccion lp ON z.id_lote_produccion = lp.id_lote_produccion
JOIN diseno d ON lp.id_diseno = d.id_diseno
WHERE z.id_molde = 1;


-- Realizar una consulta que permita conocer que lotes de material fueron usados en la construcción de un zapato.
SELECT DISTINCT 
    li.id_lote AS lote_origen_material, 
    m.nombre_material,
    li.fecha_recepcion
FROM zapato z
JOIN zapato_trozo zt ON z.id_zapato = zt.id_zapato
JOIN trozo t ON zt.id_trozo = t.id_trozo
JOIN material m ON t.id_material = m.id_material
JOIN lote_insumo li ON m.id_lote = li.id_lote
WHERE z.id_zapato = 21;


-- Realizar un consulta que permita conocer cuanto zapatos se crearon para un diseño determinado.
SELECT 
    d.nombre_diseno, 
    COUNT(z.id_zapato) AS total_zapatos_creados
FROM diseno d
JOIN lote_produccion lp ON d.id_diseno = lp.id_diseno
JOIN zapato z ON lp.id_lote_produccion = z.id_lote_produccion
WHERE d.id_diseno = 3
GROUP BY d.id_diseno, d.nombre_diseno;
